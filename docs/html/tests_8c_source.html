112void test_contexts_init() {
113 fprintf(stderr, "test_contexts_init: ");
114 // setup
115 const size_t block_size = 2339;
116 const size_t capacity = block_size + 1;
117 gf4_poly_t inv = gf4_poly_init_zero(2*capacity);
118 gf4_poly_t tmp = gf4_poly_init_zero(2*capacity);
119 gf4_poly_t div = gf4_poly_init_zero(2*capacity);
120 gf4_poly_t rem = gf4_poly_init_zero(2*capacity);
121 gf4_poly_t modulus = gf4_poly_init_zero(capacity);
122 gf4_poly_set_coefficient(&modulus, 0, 1);
123 gf4_poly_set_coefficient(&modulus, block_size, 1);
124
125 // tests
126 for (size_t i = 0; i < 10; ++i) {
127 encoding_context_t ec;
128 decoding_context_t dc;
129 contexts_init(&ec, &dc, block_size, 37);
130
131 // Test1: h1 must be invertible
132 assert(gf4_poly_invert_slow(&inv, &dc.h1, &modulus));
133
134 // Test2:
135 // H*G^T must be 0
136 // H = (H0|H1), G=(I|-(H0*H1^-1)^T)
137 // H*G^T = H0 - H1*H0*H1^-1 ---> dc.h0 - dc.h1*ec.second_block_G
138 gf4_poly_mul(&tmp, &dc.h1, &ec.second_block_G);
139 gf4_poly_div_rem(&div, &rem, &tmp, &modulus);
140 gf4_poly_add_inplace(&rem, &dc.h0);
141 assert(gf4_poly_is_zero(&rem));
142
143 gf4_poly_zero_out(&div);
144 gf4_poly_zero_out(&rem);
145 gf4_poly_zero_out(&tmp);
146 gf4_poly_zero_out(&inv);
147 contexts_deinit(&ec, &dc);
148 }
149
150 // cleanup
151 gf4_poly_deinit(&rem);
152 gf4_poly_deinit(&div);
153 gf4_poly_deinit(&inv);
154 gf4_poly_deinit(&modulus);
155 gf4_poly_deinit(&tmp);
156 print_OK();
157}
