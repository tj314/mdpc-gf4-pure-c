28gf4_matrix_t gf4_matrix_init_cyclic_matrix(gf4_poly_t * first_row, size_t N) {
29 assert(NULL != first_row);
30 assert(first_row->coefficients.capacity >= N);
31 assert(0 != N);
32 gf4_matrix_t matrix;
33 matrix.rows = malloc(N * sizeof(gf4_t *));
34 if (NULL == matrix.rows) {
35 fprintf(stderr, "gf4_matrix_init_cyclic_matrix: Memory allocation failed!\n");
36 exit(-1);
37 }
38 for (size_t row = N; row > 0; --row) {
39 matrix.rows[N - row] = malloc(N * sizeof(gf4_t));
40 if (NULL == matrix.rows[N - row]) {
41 fprintf(stderr, "gf4_matrix_init_cyclic_matrix: Memory allocation failed!\n");
42 exit(-1);
43 }
44 for (size_t col = 0; col < N; ++col) {
45 matrix.rows[N - row][col] = first_row->coefficients.array[(row + col) % N];
46 }
47 }
48 matrix.num_rows = N;
49 matrix.num_cols = N;
50 return matrix;
51}

75 // find the index of pivot in the given row
76 for (size_t col = 0; col < matrix->num_cols; ++col) {
77 if (0 != matrix->rows[row_index][col]) {
78 *out_pivot_index = col;

80 }
81 }

94bool gf4_matrix_is_upper_echelon(gf4_matrix_t * matrix) {
95 assert(NULL != matrix);
96 size_t pivot_index, last_pivot_index = 0;
97 bool last_pivot_exists = true;
98 for (size_t row = 0; row < matrix->num_rows; ++row) {
99 bool pivot_exists = gf4_matrix_find_pivot_index(matrix, row, &pivot_index);
100 if (last_pivot_exists) {
101 last_pivot_exists = pivot_exists;
102 if (last_pivot_index > pivot_index) {
103 return false;
104 }
105 for (size_t current_row = 0; current_row < matrix->num_rows; ++current_row) {
106 if (row == current_row) {
107 continue;
108 }
109 if (matrix->rows[current_row][pivot_index] != 0) {
110 return false;
111 }
112 }
113 last_pivot_index = pivot_index;
114 } else if (pivot_exists) {
115 // last row was all zeros, but this row is not
116 return false;
117 }
118 }
119 return true;
120}

144void gf4_matrix_gaussian_elimination_inplace(gf4_matrix_t * matrix) {
145 size_t * pivot_indices = calloc(matrix->num_rows, sizeof(size_t));
146 assert(NULL != pivot_indices);
147 // apply gaussian elimination from top to bottom
148 for (size_t row = 0; row < matrix->num_rows; ++row) {
149 size_t pivot_index;
150 bool pivot_found = gf4_matrix_find_pivot_index(matrix, row, &pivot_index);
151 pivot_indices[row] = pivot_index;
152 if (!pivot_found) {
153 // zero row
154 continue;
155 }
156 for (size_t other_row = 0; other_row < matrix->num_rows; ++other_row) {
157 if (row == other_row || 0 == matrix->rows[other_row][pivot_index]) {
158 continue;
159 }
160 gf4_t ratio = gf4_div(matrix->rows[row][pivot_index], matrix->rows[other_row][pivot_index]);
161 for (size_t col = 0; col < matrix->num_cols; ++ col) {
162 matrix->rows[other_row][col] = matrix->rows[row][col] ^ gf4_mul(matrix->rows[other_row][col], ratio);
163 }
164 }
165 }
166
167 // swap rows to obtain echelon form
168 size_t min_index = gf4_matrix_helper_find_min(pivot_indices, matrix->num_rows, matrix->num_cols);
169 size_t row_pos = 0;
170 while (min_index < matrix->num_rows) {
171 // swap rows in matrix
172 gf4_t * tmp_row = matrix->rows[row_pos];
173 matrix->rows[row_pos] = matrix->rows[min_index];
174 matrix->rows[min_index] = tmp_row;
175 // swap pivot indices
176 size_t tmp_index = pivot_indices[row_pos];
177 pivot_indices[row_pos] = pivot_indices[min_index];
178 pivot_indices[min_index] = tmp_index;
179 // update row_pos and min_index
180 min_index = gf4_matrix_helper_find_min(pivot_indices, matrix->num_rows, matrix->num_cols);
181 ++row_pos;
182 }
183
184 // cleanup
185 free(pivot_indices);
186}

188size_t gf4_matrix_rank(gf4_matrix_t * matrix) {
189 size_t num_zero = 0;
190 for (size_t row = 0; row < matrix->num_rows; ++row) {
191 num_zero += (size_t)gf4_matrix_is_row_zero(matrix, row);
192 }
193 assert(matrix->num_rows >= num_zero);
194 return matrix->num_rows - num_zero;
195}

197gf4_matrix_t gf4_matrix_solve_homogenous_linear_system(gf4_matrix_t * equations) {
198 assert(NULL != equations);
199 assert(gf4_matrix_is_upper_echelon(equations));
200 size_t equations_rank = gf4_matrix_rank(equations);
201 size_t num_params = equations->num_cols - equations_rank;
202
203 // allocate matrix for the solutions
204 gf4_matrix_t out_solutions;
205 out_solutions.num_cols = equations->num_cols;
206 out_solutions.num_rows = utils_binary_pow(4, num_params);
207 out_solutions.rows = malloc(out_solutions.num_rows * sizeof(gf4_t *));
208 assert(NULL != out_solutions.rows);
209 for (size_t i = 0; i < out_solutions.num_rows; ++i) {
210 out_solutions.rows[i] = calloc(out_solutions.num_cols, sizeof(gf4_t));
211 assert(NULL != out_solutions.rows[i]);
212 }
213
214 // solve
215 if (0 == equations_rank) {
216 fprintf(stderr, "gf4_matrix_solve_homogenous_linear_system: The solution to the system of equations is all permutations of length %zu. This is not yet implemented!\n", equations->num_cols);
217 exit(-1);
218 } else {
219 size_t current_row = equations_rank;
220 size_t repetitions = 1;
221 size_t last_set_unknown_index = equations->num_cols - 1;
222 size_t pivot_index;
223 do {
224 --current_row;
225 bool pivot_exists = gf4_matrix_find_pivot_index(equations, current_row, &pivot_index);
226 if (!pivot_exists) {
227 fprintf(stderr, "gf4_matrix_solve_homogenous_linear_system: Unexpected error! Encountered a zero row where no such row was expected!\n");
228 exit(-1);
229 }
230 for (size_t col = last_set_unknown_index; col > pivot_index; --col) {
231 gf4_t val = 0;
232 size_t index = 0;
233 while (index < out_solutions.num_rows) {
234 for (size_t rep = 0; rep < repetitions; ++rep) {
235 out_solutions.rows[index][col] = val;
236 ++index;
237 }
238 ++val;
239 val %= 4;
240 }
241 repetitions *= 4;
242 }
243 for (size_t row = 0; row < out_solutions.num_rows; ++row) {
244 gf4_t sum_right_of_pivot = 0;
245 gf4_t pivot_coefficient = equations->rows[current_row][pivot_index];
246 for (size_t col = pivot_index + 1; col < out_solutions.num_cols; ++col) {
247 sum_right_of_pivot ^= gf4_mul(equations->rows[current_row][col], out_solutions.rows[row][col]);
248 }
249 out_solutions.rows[row][pivot_index] = gf4_div(sum_right_of_pivot, pivot_coefficient);
250 }
251 last_set_unknown_index = (0 == pivot_index) ? equations->num_cols : last_set_unknown_index - 1;
252 } while (0 != current_row);
253
254 while (0 != pivot_index) {
255 --pivot_index;
256 gf4_t val = 0;
257 size_t index = 0;
258 while (index < out_solutions.num_rows) {
259 for (size_t rep = 0; rep < repetitions; ++rep) {
260 out_solutions.rows[index][pivot_index] = val;
261 ++index;
262 }
263 ++val;
264 val %= 4;
265 }
266 repetitions *= 4;
267 }
268 }
269 return out_solutions;
270}
