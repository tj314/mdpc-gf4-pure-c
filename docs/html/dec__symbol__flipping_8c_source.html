22bool dec_decode_symbol_flipping(gf4_array_t *maybe_decoded, gf4_array_t *in_array, size_t num_iterations, decoding_context_t * ctx) {
23 assert(NULL != maybe_decoded);
24 assert(NULL != in_array);
25 assert(NULL != ctx);
26 assert(maybe_decoded->capacity >= 2*ctx->block_size);
27 assert(in_array->capacity >= 2 * ctx->block_size);
28
29 gf4_array_t syndrome = gf4_array_init(ctx->block_size, true);
30 dec_calculate_syndrome(&syndrome, in_array, ctx);
31 memcpy(maybe_decoded->array, in_array->array, sizeof(gf4_t)*in_array->capacity);
32
33 for (size_t i = 0; i < num_iterations; ++i) {
34 long syndrome_weight = (long) gf4_array_hamming_weight(&syndrome);
35 if (0 == syndrome_weight) {
36 gf4_array_deinit(&syndrome);
37 ctx->elapsed_iterations = i;
38 return true;
39 }
40 long sigma_max = -1;
41 size_t pos = 0;
42 gf4_t a_max = 0;
43 for (size_t j = 0; j < 2*ctx->block_size; ++j) {
44 gf4_poly_t *h_block;
45 size_t actual_j;
46 if (j < ctx->block_size) {
47 h_block = &ctx->h0;
48 actual_j = j;
49 } else {
50 h_block = &ctx->h1;
51 actual_j = j - ctx->block_size;
52 }
53 for (gf4_t a = 1; a <= GF4_MAX_VALUE; ++a) {
54 // s = s - a*h_j, where h_j is j-th column of H
55 size_t x = actual_j;
56 size_t idx = 0;
57 long tmp_hamming_weight = 0; // hamming weight of s-a*h_j
58 do {
59 gf4_t tmp = syndrome.array[idx] ^ gf4_mul(h_block->coefficients.array[x], a);
60 if (0 != tmp) {
61 tmp_hamming_weight += 1;
62 }
63 x = (0 == x) ? ctx->block_size - 1: x-1;
64 ++idx;
65 } while (x != actual_j);
66
67 long sigma = syndrome_weight - tmp_hamming_weight;
68 if (sigma > sigma_max) {
69 sigma_max = sigma;
70 a_max = a;
71 pos = j;
72 }
73 }
74 }
75 // fprintf(stderr, "flipping pos=%zu with a_max=%u and sigma_max=%ld\n", pos, a_max, sigma_max);
76 gf4_poly_t *h_block;
77 size_t h_pos;
78 if (pos < ctx->block_size) {
79 h_block = &ctx->h0;
80 h_pos = pos;
81 } else {
82 h_block = &ctx->h1;
83 h_pos = pos - ctx->block_size;
84 }
85 size_t x = h_pos;
86 size_t idx = 0;
87 do {
88 syndrome.array[idx] ^= gf4_mul(h_block->coefficients.array[x], a_max);
89 x = (0 == x) ? ctx->block_size - 1: x-1;
90 ++idx;
91 } while (x != h_pos);
92
93 maybe_decoded->array[pos] ^= a_max;
94 }
95 gf4_array_deinit(&syndrome);
96 ctx->elapsed_iterations = num_iterations;
97 return false;
98}
