21void dec_calculate_syndrome(gf4_array_t *out_syndrome, gf4_array_t *in_message, decoding_context_t * ctx) {
22 assert(NULL != out_syndrome);
23 assert(NULL != in_message);
24 assert(NULL != ctx);
25 assert(out_syndrome->capacity >= ctx->block_size);
26 assert(in_message->capacity >= 2 * ctx->block_size);
27 for (size_t i = ctx->block_size; i > 0; --i) {
28 gf4_t tmp = 0;
29 for (size_t j = 0; j < ctx->block_size; ++j) {
30 tmp ^= gf4_mul(ctx->h0.coefficients.array[(i+j) % ctx->block_size], in_message->array[j]);
31 tmp ^= gf4_mul(ctx->h1.coefficients.array[(i+j) % ctx->block_size], in_message->array[ctx->block_size + j]);
32 }
33 out_syndrome->array[ctx->block_size - i] = tmp;
34 }
35}

37long dec_calculate_new_sigma(gf4_poly_t * h_block, gf4_array_t *syndrome, long syndrome_weight, gf4_t a, size_t actual_j, decoding_context_t * ctx) {
38 // s = s - a*h_j, where h_j is j-th column of H
39 size_t x = actual_j;
40 size_t idx = 0;
41 long tmp_hamming_weight = 0; // hamming weight of s-a*h_j
42 do {
43 gf4_t tmp = syndrome->array[idx] ^ gf4_mul(h_block->coefficients.array[x], a);
44 if (0 != tmp) {
45 tmp_hamming_weight += 1;
46 }
47 x = (0 == x) ? ctx->block_size - 1: x-1;
48 ++idx;
49 } while (x != actual_j);
50
51 return syndrome_weight - tmp_hamming_weight;
52}

54void dec_flip_symbol(gf4_poly_t * h_block, gf4_array_t *syndrome, gf4_array_t *maybe_decoded, gf4_t a, size_t actual_j, size_t j, decoding_context_t * ctx) {
55 size_t x = actual_j;
56 size_t idx = 0;
57 do {
58 syndrome->array[idx] ^= gf4_mul(h_block->coefficients.array[x], a);
59 x = (0 == x) ? ctx->block_size - 1: x-1;
60 ++idx;
61 } while (x != actual_j);
62 maybe_decoded->array[j] ^= a;
63}

66 bool decoding_success = decode(out_decrypted, in_encrypted, num_iterations, ctx);
