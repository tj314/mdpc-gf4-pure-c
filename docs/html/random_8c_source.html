35size_t random_from_range(size_t low_bound_inclusive, size_t top_bound_inclusive) {
36 assert(low_bound_inclusive < top_bound_inclusive);
37 random_init();
38 return low_bound_inclusive + (rand() % (top_bound_inclusive - low_bound_inclusive + 1));
39}

49void random_weighted_gf4_array(gf4_array_t *array, size_t size, size_t weight) {
50 assert(NULL != array);
51 assert(size <= array->capacity);
52 assert(weight <= size);
53 // place weight nonzero entries on the first weight positions
54 for (size_t i = 0; i < weight; ++i) {
55 array->array[i] = (gf4_t) random_from_range(1, GF4_MAX_VALUE);
56 }
57 // shuffle
58 for (size_t i = 0; i < size - 1; ++i) {
59 size_t j = (size_t)random_from_range(i, size - 1);
60 gf4_t tmp = array->array[i];
61 array->array[i] = array->array[j];
62 array->array[j] = tmp;
63 }
64}

66void random_weighted_gf4_vector_pairs_common(gf4_array_t *vector, size_t size, size_t weight, size_t distance, gf4_t first, gf4_t second) {
67 assert(NULL != vector);
68 assert(vector->capacity >= size);
69 assert(size > weight);
70 size_t num_pairs = weight / 2;
71 for (size_t i = 0; i < num_pairs; ++i) {
72 size_t index = random_from_range(0, size - 1);
73 size_t second_index = (index + distance) % size;
74 while (vector->array[index] > 0 || vector->array[second_index] > 0) {
75 index = random_from_range(0, size - 1);
76 second_index = (index + distance) % size;
77 }
78 vector->array[index] = first;
79 vector->array[second_index] = second;
80 }
81}
