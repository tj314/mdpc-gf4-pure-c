63bool dec_decode_symbol_flipping_threshold(gf4_array_t *maybe_decoded, gf4_array_t *in_array, size_t num_iterations, decoding_context_t * ctx) {
64 assert(NULL != maybe_decoded);
65 assert(NULL != in_array);
66 assert(NULL != ctx);
67 assert(maybe_decoded->capacity >= 2*ctx->block_size);
68 assert(in_array->capacity >= 2 * ctx->block_size);
69 assert(NULL != ctx->threshold);
70
71 gf4_array_t syndrome = gf4_array_init(ctx->block_size, true);
72 dec_calculate_syndrome(&syndrome, in_array, ctx);
73 memcpy(maybe_decoded->array, in_array->array, sizeof(gf4_t)*in_array->capacity);
74
75 for (size_t i = 0; i < num_iterations; ++i) {
76 long syndrome_weight = (long) gf4_array_hamming_weight(&syndrome);
77 if (0 == syndrome_weight) {
78 gf4_array_deinit(&syndrome);
79 ctx->elapsed_iterations = i;
80 return true;
81 }
82 long threshold = ctx->threshold(syndrome_weight);
83 for (size_t j = 0; j < 2*ctx->block_size; ++j) {
84 gf4_poly_t *h_block;
85 size_t actual_j;
86 if (j < ctx->block_size) {
87 h_block = &ctx->h0;
88 actual_j = j;
89 } else {
90 h_block = &ctx->h1;
91 actual_j = j - ctx->block_size;
92 }
93 long sigma_max = -1;
94 gf4_t a_max = 0;
95 for (gf4_t a = 1; a <= GF4_MAX_VALUE; ++a) {
96 long sigma = dec_calculate_new_sigma(h_block, &syndrome, syndrome_weight, a, actual_j, ctx);
97 if (sigma > sigma_max) {
98 sigma_max = sigma;
99 a_max = a;
100 }
101 }
102
103 if (sigma_max > threshold) {
104 maybe_decoded->array[j] ^= a_max;
105 }
106 }
107 gf4_array_zero_out(&syndrome);
108 dec_calculate_syndrome(&syndrome, maybe_decoded, ctx);
109 }
110 gf4_array_deinit(&syndrome);
111 ctx->elapsed_iterations = num_iterations;
112 return false;
113}
