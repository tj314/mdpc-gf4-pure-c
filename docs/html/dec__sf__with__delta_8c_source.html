21bool dec_decode_symbol_flipping_delta(gf4_array_t *maybe_decoded, gf4_array_t *in_array, size_t num_iterations, decoding_context_t * ctx) {
22 assert(NULL != maybe_decoded);
23 assert(NULL != in_array);
24 assert(NULL != ctx);
25 assert(maybe_decoded->capacity >= 2*ctx->block_size);
26 assert(in_array->capacity >= 2 * ctx->block_size);
27 assert(ctx->delta_setting >= 0);
28
29 gf4_array_t syndrome = gf4_array_init(ctx->block_size, true);
30 dec_calculate_syndrome(&syndrome, in_array, ctx);
31 memcpy(maybe_decoded->array, in_array->array, sizeof(gf4_t)*in_array->capacity);
32 ctx->elapsed_iterations = 0;
33
34 long * sigmas = calloc(2 * ctx->block_size, sizeof(long));
35 assert(NULL != sigmas);
36 gf4_t * values = calloc(2 * ctx->block_size, sizeof(gf4_t));
37 assert(NULL != values);
38
39 const long DELTA = ctx->delta_setting;
40
41 for (size_t i = 0; i < num_iterations; ++i) {
42 long syndrome_weight = (long) gf4_array_hamming_weight(&syndrome);
43 if (0 == syndrome_weight) {
44 free(sigmas);
45 free(values);
46 gf4_array_deinit(&syndrome);
47 ctx->elapsed_iterations = i;
48 return true;
49 }
50 long sigma_max = -1;
51 for (size_t j = 0; j < 2*ctx->block_size; ++j) {
52 sigmas[j] = -1;
53 values[j] = 0;
54 gf4_poly_t *h_block;
55 size_t actual_j;
56 if (j < ctx->block_size) {
57 h_block = &ctx->h0;
58 actual_j = j;
59 } else {
60 h_block = &ctx->h1;
61 actual_j = j - ctx->block_size;
62 }
63 for (gf4_t a = 1; a <= GF4_MAX_VALUE; ++a) {
64 // s = s - a*h_j, where h_j is j-th column of H
65 size_t x = actual_j;
66 size_t idx = 0;
67 long new_syndrome_weight = 0; // hamming weight of s-a*h_j
68 do {
69 gf4_t tmp = syndrome.array[idx] ^ gf4_mul(h_block->coefficients.array[x], a);
70 if (0 != tmp) {
71 new_syndrome_weight += 1;
72 }
73 x = (0 == x) ? ctx->block_size - 1 : x-1;
74 ++idx;
75 } while (x != actual_j);
76
77 long sigma = syndrome_weight - new_syndrome_weight;
78 if (sigma > sigma_max) {
79 sigma_max = sigma;
80 }
81 if (sigma > sigmas[j]) {
82 sigmas[j] = sigma;
83 values[j] = a;
84 }
85 }
86 }
87
88 long bound = ((sigma_max - DELTA) >= 0) ? sigma_max - DELTA : 0;
89 for (size_t j = 0; j < 2*ctx->block_size; ++j) {
90 if (sigmas[j] < bound) continue;
91 gf4_poly_t *h_block;
92 size_t actual_j;
93 if (j < ctx->block_size) {
94 h_block = &ctx->h0;
95 actual_j = j;
96 } else {
97 h_block = &ctx->h1;
98 actual_j = j - ctx->block_size;
99 }
100 size_t x = actual_j;
101 size_t idx = 0;
102 do {
103 syndrome.array[idx] ^= gf4_mul(h_block->coefficients.array[x], values[j]);
104 x = (0 == x) ? ctx->block_size - 1 : x-1;
105 ++idx;
106 } while (x != actual_j);
107 maybe_decoded->array[j] ^= values[j];
108 }
109 }
110 free(sigmas);
111 free(values);
112 gf4_array_deinit(&syndrome);
113 ctx->elapsed_iterations = num_iterations;
114 return false;
