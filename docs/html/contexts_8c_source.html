5gf4_t poly_sum(gf4_poly_t * poly) {
6 assert(NULL != poly);
7 gf4_t sum = 0;
8 for (size_t i = 0; i <= poly->degree; ++i) {
9 sum ^= poly->coefficients[i];
10 }
