28gf4_array_t gf4_array_init(size_t capacity, bool zero_out_new_memory) {
29 assert(0 < capacity);
30 gf4_array_t out;
31 if (zero_out_new_memory) {
32 out.array = calloc(capacity, sizeof(gf4_t));
33 } else {
34 out.array = malloc(capacity * sizeof(gf4_t));
35 }
36 if (NULL == out.array) {
37 fprintf(stderr, "%s: Allocation error!\n", __func__);
38 exit(-1);
39 }
40 out.capacity = capacity;
41 return out;
42}

56void gf4_array_resize(gf4_array_t * array, size_t new_capacity, bool zero_out_new_memory) {
57 assert(NULL != array);
58 assert(0 < new_capacity);
59 gf4_t * tmp = realloc(array->array, new_capacity * sizeof(gf4_t));
60 if (NULL == tmp) {
61 free(array->array);
62 fprintf(stderr, "gf4_array_resize: Memory reallocation failed!\n");
63 exit(-1);
64 }
65 if (zero_out_new_memory) {
66 for (size_t i = array->capacity; i < new_capacity; ++i) {
67 tmp[i] = 0;
68 }
69 }
70 array->array = tmp;
71 array->capacity = new_capacity;
72}
