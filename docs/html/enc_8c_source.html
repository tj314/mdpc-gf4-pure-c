22void enc_encode(gf4_array_t *out_encoded, gf4_array_t *in_message, encoding_context_t * ctx) {
23 assert(NULL != out_encoded);
24 assert(NULL != in_message);
25 assert(NULL != ctx);
26 assert(out_encoded->capacity >= 2*ctx->block_size);
27 assert(in_message->capacity >= ctx->block_size);
28 memcpy(out_encoded->array, in_message->array, ctx->block_size);
29 for (size_t i = ctx->block_size; i > 0; --i) {
30 gf4_t tmp = 0;
31 for (size_t j = 0; j < ctx->block_size; ++j) {
32 tmp ^= gf4_mul(in_message->array[j], ctx->second_block_G.coefficients.array[(i + j) % ctx->block_size]);
33 }
34 out_encoded->array[2*ctx->block_size - i] = tmp;
35 }
36}

38void enc_encrypt(gf4_array_t *out_encrypted, gf4_array_t *in_message, size_t num_errors, encoding_context_t * ctx) {
39 assert(NULL != out_encrypted);
40 assert(NULL != in_message);
41 assert(NULL != ctx);
42 assert(out_encrypted->capacity >= 2*ctx->block_size);
43 assert(in_message->capacity >= ctx->block_size);
44 enc_encode(out_encrypted, in_message, ctx);
45 gf4_array_t err = gf4_array_init(2*ctx->block_size, true);
46 random_weighted_gf4_array(&err, 2 * ctx->block_size, num_errors);
47#ifdef WRITE_WEIGHTS
48 char fname[100] = {0};
49 sprintf(fname, "errorvec-exp_%zu.txt", ctx->index);
50 FILE * outfile = fopen(fname, "w");

52 fprintf(outfile, "%u;", err.coefficients.array[i]);

54 fprintf(outfile, "\n");
55 fclose(outfile);
56#endif
57 for (size_t i = 0; i < 2*ctx->block_size; ++i) {
58 out_encrypted->array[i] = gf4_add(out_encrypted->array[i], err.array[i]);
59 }
60 gf4_array_deinit(&err);
61}
